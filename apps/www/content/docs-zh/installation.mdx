import { Step, Steps } from 'fumadocs-ui/components/steps';
import { CodeTabs } from '@/components/docs/code-tabs';

<Callout>
  **提示：** JetBrains UI 使用与
  [shadcn/ui](https://ui.shadcn.com/docs/installation) 一致的安装流程。
</Callout>

<Steps>
<Step>

### 初始化 shadcn/ui

执行 `init` 命令来创建带有 shadcn/ui 的新项目，或初始化已有项目。

<CodeTabs
  codes={{
    pnpm: `pnpm dlx shadcn@latest init`,
    npm: `npx shadcn@latest init`,
    yarn: `yarn shadcn@latest init`,
    bun: `bunx --bun shadcn@latest init`,
  }}
/>

</Step>
<Step>

### 添加组件

执行 `add` 命令将组件添加到你的项目中。

<CodeTabs
  codes={{
    pnpm: `pnpm dlx shadcn@latest add @jetbrains-ui/buttons`,
    npm: `npx shadcn@latest add @jetbrains-ui/buttons`,
    yarn: `npx shadcn@latest add @jetbrains-ui/buttons`,
    bun: `bunx --bun shadcn@latest add @jetbrains-ui/buttons`,
  }}
/>

</Step>
<Step>

### 导入组件

在项目中导入并使用组件。

```tsx
import { Button } from '@/components/jetbrains-ui/buttons';

export default function Home() {
  return (
    <div>
      <Button>Click me</Button>
    </div>
  );
}
```

</Step>
</Steps>
