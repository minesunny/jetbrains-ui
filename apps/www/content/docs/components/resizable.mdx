## Overview

A composable resizable layout component with support for collapsible panels and remembered size.

`collapsedSize` defaults to `0`. Use `-1` to disable collapse. Use `collapsed` to control the current collapsed state.

## Installation

<ComponentInstallation name="resizable" />

## Usage

```tsx
import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from '@/components/jetbrains-ui/resizable';

<ResizablePanelGroup direction="horizontal">
  <ResizablePanel defaultSize={40} minSize={20} collapsedSize={0}>
    A
  </ResizablePanel>
  <ResizableHandle />
  <ResizablePanel defaultSize={60} minSize={30}>
    B
  </ResizablePanel>
</ResizablePanelGroup>;
```

## Use Case: Collapsible Panel

```tsx
'use client';

import * as React from 'react';

import {
  ResizableHandle,
  ResizablePanel,
  ResizablePanelGroup,
} from '@/components/jetbrains-ui/resizable';

export function ResizableCollapsibleExample() {
  const [collapsed, setCollapsed] = React.useState(false);

  return (
    <div className="space-y-3">
      <button
        type="button"
        className="rounded border px-3 py-1 text-sm"
        onClick={() => setCollapsed((prev) => !prev)}
      >
        Toggle Panel A
      </button>

      <ResizablePanelGroup
        direction="horizontal"
        className="h-56 rounded-lg border"
      >
        <ResizablePanel
          defaultSize={45}
          minSize={20}
          collapsedSize={0}
          collapsed={collapsed}
          onCollapse={() => setCollapsed(true)}
          onExpand={() => setCollapsed(false)}
        >
          <div className="h-full p-3">Panel A</div>
        </ResizablePanel>
        <ResizableHandle withHandle />
        <ResizablePanel defaultSize={55} minSize={25}>
          <div className="h-full p-3">Panel B</div>
        </ResizablePanel>
      </ResizablePanelGroup>
    </div>
  );
}
```
