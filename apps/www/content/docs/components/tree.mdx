## Overview

A JetBrains-style tree component for hierarchical data display.

## Installation

<ComponentInstallation name="tree" />

## Usage

```tsx
import { Tree, type TreeItemData } from '@/components/jetbrains-ui/tree';

const items: TreeItemData[] = [
  {
    id: 'src',
    label: 'src',
    children: [{ id: 'src-index', label: 'index.tsx' }],
  },
];

<Tree items={items} defaultExpandedIds={['src']} />;
```
