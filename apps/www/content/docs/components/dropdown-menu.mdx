## Overview

A JetBrains-style dropdown menu component.

## Installation

<ComponentInstallation name="dropdown-menu" />

## Usage

```tsx
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuTrigger,
} from '@/components/jetbrains-ui/dropdown-menu';
import { Button } from '@/components/jetbrains-ui/buttons';

<DropdownMenu>
  <DropdownMenuTrigger asChild>
    <Button>Open Menu</Button>
  </DropdownMenuTrigger>
  <DropdownMenuContent>
    <DropdownMenuItem>Profile</DropdownMenuItem>
  </DropdownMenuContent>
</DropdownMenu>;
```
